  WHEN OTHERS THEN
       RETURN;

/* ファイル選択ダイアログ */
FUNCTION OPEN_FILE_DIALOG RETURN VARCHAR2 IS
  W_DIALOG OLE2.OBJ_TYPE;
  W_PATH   VARCHAR2(300);
  PROCEDURE QUIT_EXCEL IS
  BEGIN
  	IF W_DIALOG IS NOT NULL THEN
  		OLE2.INVOKE(W_DIALOG, 'Quit'); 
  		OLE2.RELEASE_OBJ(W_DIALOG);
  	END IF;
  EXCEPTION
    WHEN OTHERS THEN
      NULL;		
  END;
BEGIN
	-- 初期化
	W_PATH := '';

  BEGIN
    W_DIALOG := OLE2.CREATE_OBJ('Excel.Application');
    OLE2.SET_PROPERTY(W_DIALOG, 'Visible', 'False');
    OLE2.SET_PROPERTY(W_DIALOG, 'DisplayAlerts', 'False');
    W_PATH := OLE2.INVOKE_CHAR(W_DIALOG, 'GetOpenFilename');
    QUIT_EXCEL;
    IF W_PATH = '0' THEN
    	W_PATH := '';
    END IF;
    RETURN W_PATH;
  EXCEPTION
    WHEN OTHERS THEN
      QUIT_EXCEL;
  END;
  
  BEGIN
    W_PATH := GET_FILE_NAME();
    RETURN W_PATH;
  EXCEPTION
    WHEN OTHERS THEN
      NULL;
  END;
  RETURN W_PATH;
EXCEPTION
  WHEN OTHERS THEN
    RETURN '';
END;
